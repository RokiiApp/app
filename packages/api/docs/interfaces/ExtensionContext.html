Interface ExtensionContext

The context that is provided to the extension when it is executed. This context provides the extension with the input term, and a function that can be called to display the results of the extension.

interface ExtensionContext {
    actions: {
        copyToClipboard: ((text) => Promise<void>);
        hideWindow: (() => Promise<void>);
        open: ((url) => Promise<void>);
        replaceTerm: ((term) => void);
        reveal: ((path) => Promise<void>);
    };
    display: ((items) => Promise<void>);
    hide: ((id) => void);
    settings: Record<string, any>;
    term: string;
    update: ((id, item) => void);
}
